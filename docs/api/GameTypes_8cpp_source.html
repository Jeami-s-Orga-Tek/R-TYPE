206void GameManager::handleEvents(sf::RenderWindow& window)
207{
208 sf::Event event;
209 while (window.pollEvent(event)) {
210 if (event.type == sf::Event::Closed) {
211 currentState = State::QUIT;
212 }
213 if (event.type == sf::Event::KeyPressed) {
214 handleKeyPress(event, window);
215 }
216 if (event.type == sf::Event::TextEntered && isEditingUsername) {
217 if (event.text.unicode == '\b') {
218 if (!UsernameGame.empty() && cursorPos > 0) {
219 UsernameGame.erase(cursorPos - 1, 1);
220 cursorPos--;
221 }
222 }
223 else if (event.text.unicode == 127) {
224 if (!UsernameGame.empty() && cursorPos < UsernameGame.size()) {
225 UsernameGame.erase(cursorPos, 1);
226 }
227 }
228 else if (event.text.unicode == '\r' || event.text.unicode == '\n') {
229 std::ofstream file("Username.txt");
230 if (file.is_open()) {
231 file << UsernameGame;
232 file.close();
233 }
234 username = Username(username.getSize(), username.getSize(), UsernameGame, font);
235 isEditingUsername = false;
236 }
237 else if (event.text.unicode < 128 && std::isprint(event.text.unicode)) {
238 UsernameGame.insert(cursorPos, 1, static_cast<char>(event.text.unicode));
239 cursorPos++;
240 }
241 std::ofstream file("Username.txt");
242 if (file.is_open()) {
243 file << UsernameGame;
244 file.close();
245 }
246 }
247 if (event.type == sf::Event::KeyPressed && isEditingUsername) {
248 if (event.key.code == sf::Keyboard::Left && cursorPos > 0) {
249 cursorPos--;
250 }
251 if (event.key.code == sf::Keyboard::Right && cursorPos < UsernameGame.size()) {
252 cursorPos++;
253 }
254 if (event.key.code == sf::Keyboard::Delete) {
255 if (!UsernameGame.empty() && cursorPos < UsernameGame.size()) {
256 UsernameGame.erase(cursorPos, 1);
257 }
258 std::ofstream file("Username.txt");
259 if (file.is_open()) {
260 file << UsernameGame;
261 file.close();
262 }
263 }
264 }
265 if (event.type == sf::Event::Resized) {
266 sf::FloatRect visibleArea(0, 0, event.size.width, event.size.height);
267 window.setView(sf::View(visibleArea));
268 handleWindowResize(event);
269 }
270 if (event.type == sf::Event::MouseButtonPressed) {
271 handleMouseClick(event, window);
272 }
273 if (event.type == sf::Event::MouseButtonReleased) {
274 if (event.mouseButton.button == sf::Mouse::Left) {
275 isDraggingVolume = false;
276 }
277 }
278 if (event.type == sf::Event::MouseMoved) {
279 handleMouseMove(window);
280 } else {
281 paramButton.setHovered(false);
282 fps30Button.setHovered(false);
283 fps60Button.setHovered(false);
284 backButton.setHovered(false);
285 }
286 launch.handleEvent(event, window);
287 if (currentState == State::CONTROLS) {
288 controlsConfig.handleEvent(event, window);
289 }
290 lobby.handleEvent(event, window);
291 player.handleEvent(event, window);
292 errorServer.handleEvent(event, window);
293 }
294}

296void GameManager::update()
297{
298 float deltaTime = deltaClock.restart().asSeconds();
299 float coinTime = insertCoinClock.getElapsedTime().asSeconds();
300
301 if (currentState == State::LAUNCH) {
302 coinTime = static_cast<int>(128 + 127 * std::sin(coinTime * 2.5f));
303 insertCoinText.setFillColor(sf::Color(255, 255, 0, coinTime));
304 launch.update();
305 } else if (currentState == State::MENU) {
306 player.update();
307 } else if (currentState == State::LOCKER) {
308 player.update();
309 } else if (currentState == State::LOBBY) {
310 if ((waitingPlayersCounter == 1 && gameMode == GameMode::SOLO) || (waitingPlayersCounter == 2 && gameMode == GameMode::DUO) ||
311 (waitingPlayersCounter == 3 && gameMode == GameMode::TRIO) || (waitingPlayersCounter == 4 && gameMode == GameMode::SQUAD)) {
312 currentState = State::GAME;
313 }
314 player.update();
315 } else if (currentState == State::ERRORSERVER) {
316 errorServer.update();
317 }
318 if (currentState != State::LAUNCH) {
319 fpsDisplay.setString("FPS: " + std::to_string(currentFps));
320 }
321 particleSystem.update(deltaTime);
322}

899
900 renderer = std::make_shared<Engine::Renderers::SFML>();

915 render_system->addSprite(renderer, "ground_enemy", "ground_enemy_sprite_sheet", {33, 33}, {0, 0}, 1, 1);
916
917 const float FIXED_DT = 1.0f / 60.0f;
918 float accumulator = 0.0f;
919 auto previousTime = std::chrono::high_resolution_clock::now();
920
921 int frame_count = 0;
922 float fps = 0.0f;
923 float fps_timer = 0.0f;
924
925 renderer->loadFont("basic", "assets/r-type.otf");
926
927 while (renderer->isWindowOpen()) {
928 renderer->clearWindow();
929
930 renderer->handleEvents(networkManager);
931
932 auto currentTime = std::chrono::high_resolution_clock::now();
933 float frameTime = std::chrono::duration<float>(currentTime - previousTime).count();
934 previousTime = currentTime;
935 accumulator += frameTime;
936
937 frame_count++;
938 fps_timer += frameTime;
939 if (fps_timer >= 1.0f) {
940 fps = frame_count / fps_timer;
941 frame_count = 0;
942 fps_timer = 0.0f;
943 }
944 renderer->drawText("basic", std::to_string(mediator->getEntityCount()) + " entites pour FPS " + std::to_string((int)(fps)), 0.0f, 0.0f, 20, 0x00FF00FF);
945
946 while (accumulator >= FIXED_DT) {
947 player_control_system->update(networkManager, FIXED_DT);
948 physics_system->update(mediator, FIXED_DT);
949 collision_system->update(networkManager);
950 enemy_system->update(networkManager, FIXED_DT);
951 render_system->update(renderer, mediator);
952 accumulator -= FIXED_DT;
953 }
954
955 renderer->displayWindow();
956 }
