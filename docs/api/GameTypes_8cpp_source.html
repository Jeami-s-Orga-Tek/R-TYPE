169void GameManager::handleEvents(sf::RenderWindow& window)
170{
171 sf::Event event;
172 while (window.pollEvent(event)) {
173 if (event.type == sf::Event::Closed) {
174 currentState = State::QUIT;
175 }
176 if (event.type == sf::Event::KeyPressed) {
177 handleKeyPress(event, window);
178 }
179 if (event.type == sf::Event::Resized) {
180 sf::FloatRect visibleArea(0, 0, event.size.width, event.size.height);
181 window.setView(sf::View(visibleArea));
182 handleWindowResize(event);
183 }
184 if (event.type == sf::Event::MouseButtonPressed) {
185 handleMouseClick(event, window);
186 }
187 if (event.type == sf::Event::MouseButtonReleased) {
188 if (event.mouseButton.button == sf::Mouse::Left) {
189 isDraggingVolume = false;
190 }
191 }
192 if (event.type == sf::Event::MouseMoved) {
193 handleMouseMove(window);
194 } else {
195 paramButton.setHovered(false);
196 fps30Button.setHovered(false);
197 fps60Button.setHovered(false);
198 backButton.setHovered(false);
199 }
200 launch.handleEvent(event, window);
201 if (currentState == State::CONTROLS) {
202 controlsConfig.handleEvent(event, window);
203 }
204 lobby.handleEvent(event, window);
205 player.handleEvent(event, window);
206 errorServer.handleEvent(event, window);
207 }
208}

210void GameManager::update()
211{
212 float deltaTime = deltaClock.restart().asSeconds();
213 float coinTime = insertCoinClock.getElapsedTime().asSeconds();
214
215 if (currentState == State::LAUNCH) {
216 coinTime = static_cast<int>(128 + 127 * std::sin(coinTime * 2.5f));
217 insertCoinText.setFillColor(sf::Color(255, 255, 0, coinTime));
218 launch.update();
219 } else if (currentState == State::MENU) {
220 player.update();
221 } else if (currentState == State::LOCKER) {
222 player.update();
223 } else if (currentState == State::LOBBY) {
224 if ((waitingPlayersCounter == 1 && gameMode == GameMode::SOLO) || (waitingPlayersCounter == 2 && gameMode == GameMode::DUO) ||
225 (waitingPlayersCounter == 3 && gameMode == GameMode::TRIO) || (waitingPlayersCounter == 4 && gameMode == GameMode::SQUAD)) {
226 currentState = State::GAME;
227 }
228 player.update();
229 } else if (currentState == State::ERRORSERVER) {
230 errorServer.update();
231 }
232 if (currentState != State::LAUNCH) {
233 fpsDisplay.setString("FPS: " + std::to_string(currentFps));
234 }
235 particleSystem.update(deltaTime);
236}

589 udp::endpoint server_endpoint = *endpoints.begin();

592 std::string message = "CONNECT";

599 return false;

602 socket.non_blocking(true);
603 char buffer[1024];
604 udp::endpoint sender_endpoint;
605 auto start_time = std::chrono::steady_clock::now();
606 const auto timeout = std::chrono::seconds(5);
607
608 while (std::chrono::steady_clock::now() - start_time < timeout) {
609 boost::system::error_code receive_error;
610 size_t bytes_received = socket.receive_from(
611 boost::asio::buffer(buffer),
612 sender_endpoint,
613 0,
614 receive_error
615 );
616 if (!receive_error && bytes_received > 0) {
617 std::string response(buffer, bytes_received);
618 std::cout << "Server response: " << response << std::endl;
619 socket.close();
620 return true;
621 } else if (receive_error != boost::asio::error::would_block) {
622 std::cerr << "Error receiving: " << receive_error.message() << std::endl;
