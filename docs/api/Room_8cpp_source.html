20bool Room::join(std::shared_ptr<ClientSession> client)
21{
22 if (!client) {
23 warnf("Attempted to add null client to room '%s'", m_name.c_str());
24 return false;
25 }
26 if (isFull()) {
27 warnf("Room '%s' is full (%zu/%zu players)", m_name.c_str(), m_clients.size(), m_maxPlayers);
28 return false;
29 }
30 auto it = std::find(m_clients.begin(), m_clients.end(), client);
31 if (it != m_clients.end()) {
32 warnf("Client (playerId=%u) is already in room '%s'", client->playerId, m_name.c_str());
33 return false;
34 }
35 if (client->currentRoom && client->currentRoom.get() != this) {
36 client->currentRoom->leave(client);
37 }
38 m_clients.push_back(client);
39 infof("Client (playerId=%u) joined room '%s' (%zu/%zu players)",
40 client->playerId, m_name.c_str(), m_clients.size(), m_maxPlayers);
41 return true;
42}

44void Room::leave(std::shared_ptr<ClientSession> client)
45{
46 if (!client) return;
47 auto it = std::find(m_clients.begin(), m_clients.end(), client);
48 if (it != m_clients.end()) {

56void Room::broadcast(const std::vector<uint8_t>& message, std::shared_ptr<ClientSession> excludeClient)
57{
58 if (!m_socket) {
59 warnf("Cannot broadcast to room '%s': no socket set", m_name.c_str());
60 return;
61 }
62 size_t sentCount = 0;
63 for (const auto& client : m_clients) {
64 if (client && client != excludeClient) {
65 auto buf = std::make_shared<std::vector<uint8_t>>(message);
66 m_socket->async_send_to(
67 boost::asio::buffer(*buf), client->endpoint,
68 [buf, client, this](boost::system::error_code ec, std::size_t) {
69 if (ec) {
70 auto addr = client->endpoint.address().to_string();
71 warnf("Failed to send message to client (playerId=%u) at %s:%u in room '%s': %s",
72 client->playerId, addr.c_str(), client->endpoint.port(),
73 m_name.c_str(), ec.message().c_str());
74 }
75 });
76 sentCount++;
77 }
78 }
79 if (sentCount > 0) {
80 infof("Broadcasted message to %zu clients in room '%s'", sentCount, m_name.c_str());
81 }
82}
