55void Engine::Renderers::SFML::handleEvents(std::shared_ptr<Engine::NetworkManager> networkManager)
56{
57 std::bitset<5> buttons {};
58 sf::Event event;
59
60 while (window->pollEvent(event)) {
61 if (event.type == sf::Event::Closed)
62 window->close();
63 if (event.type == sf::Event::KeyPressed) {
64 if (event.key.code == sf::Keyboard::Space && window->hasFocus()) {
65 buttons.set(static_cast<std::size_t>(Engine::InputButtons::SHOOT));
66 }
67 }
68 }
69
70 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Left) || sf::Keyboard::isKeyPressed(sf::Keyboard::Q))) {
71 buttons.set(static_cast<std::size_t>(Engine::InputButtons::LEFT));
72 }
73 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Right) || sf::Keyboard::isKeyPressed(sf::Keyboard::D))) {
74 buttons.set(static_cast<std::size_t>(Engine::InputButtons::RIGHT));
75 }
76 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Up) || sf::Keyboard::isKeyPressed(sf::Keyboard::Z))) {
77 buttons.set(static_cast<std::size_t>(Engine::InputButtons::UP));
78 }
79 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Down) || sf::Keyboard::isKeyPressed(sf::Keyboard::S))) {
80 buttons.set(static_cast<std::size_t>(Engine::InputButtons::DOWN));
81 }
82
83 Engine::Event player_input_event(static_cast<Engine::EventId>(Engine::EventsIds::PLAYER_INPUT));
84 player_input_event.setParam(0, networkManager->player_id);
85 player_input_event.setParam(1, buttons);
86 networkManager->mediator->sendEvent(player_input_event);
87 networkManager->sendInput(networkManager->player_id, networkManager->room_id, buttons);
88}
