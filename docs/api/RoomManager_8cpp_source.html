19std::shared_ptr<Room> RoomManager::createRoom(const std::string& name, size_t maxPlayers)
20{
21 std::lock_guard<std::mutex> lock(m_mutex);
22 if (m_roomsByName.find(name) != m_roomsByName.end()) {

52bool RoomManager::removeRoom(const std::string& name)
53{
54 std::lock_guard<std::mutex> lock(m_mutex);
55
56 auto it = m_roomsByName.find(name);
57 if (it == m_roomsByName.end()) {
58 return false;
59 }
60 uint32_t roomId = it->second->getRoomId();
61 auto clients = it->second->getClients();
62 for (auto& client : clients) {
63 it->second->leave(client);

65 m_roomsByName.erase(it);

114void RoomManager::cleanup()
115{
116 std::lock_guard<std::mutex> lock(m_mutex);
117 std::vector<std::string> toRemove;
118
119 for (const auto& [name, room] : m_roomsByName) {
120 if (room->isEmpty()) {
121 toRemove.push_back(name);
122 }
123 }
124 for (const auto& name : toRemove) {
125 auto it = m_roomsByName.find(name);
126 if (it != m_roomsByName.end()) {
127 uint32_t roomId = it->second->getRoomId();
128 m_roomsById.erase(roomId);
129 m_roomsByName.erase(it);
130 infof("Auto-removed empty room '%s'", name.c_str());
131 }
132 }
