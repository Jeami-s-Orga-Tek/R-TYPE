99void ControlsConfig::handleEvent(const sf::Event& event, sf::RenderWindow& window)
100{
101 if (event.type == sf::Event::KeyPressed) {
102 if (waitingForKeyInput) {
103 if (event.key.code != sf::Keyboard::Escape) {
104 setControlKey(currentActionBeingSet, event.key.code);
105 waitingForKeyInput = false;
106 int index = static_cast<int>(currentActionBeingSet);
107 if (index >= 0 && index < 5) {
108 ControlValues[index].setFillColor(sf::Color::Yellow);
109 }
110 } else {
111 cancelKeyCapture();
112 }
113 }
114 }
115 if (event.type == sf::Event::MouseButtonPressed && event.mouseButton.button == sf::Mouse::Left) {
116 sf::Vector2i mousePos(event.mouseButton.x, event.mouseButton.y);
117 ControlAction action;
118 if (isControlButtonClicked(mousePos, action)) {
119 startKeyCapture(action);
120 }
121 }
122 if (event.type == sf::Event::Resized) {
123 windowSize.x = event.size.width;
124 windowSize.y = event.size.height;
125 updateControlSettingsPositions();
126 }
127}
