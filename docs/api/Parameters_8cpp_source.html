261sf::Color Parameters::applyColorBlindFilter(const sf::Color& originalColor) const
262{
263 switch (currentColorBlindMode) {
264 case ColorBlindMode::NORMAL:
265 return originalColor;
266
267 case ColorBlindMode::PROTANOPIA:
268 {
269 float r = originalColor.r / 255.0f;
270 float g = originalColor.g / 255.0f;
271 float b = originalColor.b / 255.0f;
272 float newR = 0.567f * r + 0.433f * g + 0.0f * b;
273 float newG = 0.558f * r + 0.442f * g + 0.0f * b;
274 float newB = 0.0f * r + 0.242f * g + 0.758f * b;
275
276 return sf::Color(
277 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newR * 255.0f))),
278 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newG * 255.0f))),
279 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newB * 255.0f))),
280 originalColor.a
281 );
282 }
283
284 case ColorBlindMode::DEUTERANOPIA:
285 {
286 float r = originalColor.r / 255.0f;
287 float g = originalColor.g / 255.0f;
288 float b = originalColor.b / 255.0f;
289 float newR = 0.625f * r + 0.375f * g + 0.0f * b;
290 float newG = 0.7f * r + 0.3f * g + 0.0f * b;
291 float newB = 0.0f * r + 0.3f * g + 0.7f * b;
292
293 return sf::Color(
294 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newR * 255.0f))),
295 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newG * 255.0f))),
296 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newB * 255.0f))),
297 originalColor.a
298 );
299 }
300 case ColorBlindMode::TRITANOPIA:
301 {
302 float r = originalColor.r / 255.0f;
303 float g = originalColor.g / 255.0f;
304 float b = originalColor.b / 255.0f;
305 float newR = 0.95f * r + 0.05f * g + 0.0f * b;
306 float newG = 0.0f * r + 0.433f * g + 0.567f * b;
307 float newB = 0.0f * r + 0.475f * g + 0.525f * b;
308
309 return sf::Color(
310 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newR * 255.0f))),
311 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newG * 255.0f))),
312 (sf::Uint8)(std::min(255.0f, std::max(0.0f, newB * 255.0f))),
313 originalColor.a
314 );
315 }
316 case ColorBlindMode::MONOCHROME:
317 {
318 float luminance = 0.299f * originalColor.r + 0.587f * originalColor.g + 0.114f * originalColor.b;
319 sf::Uint8 gray = (sf::Uint8)(std::min(255.0f, std::max(0.0f, luminance)));
320
321 return sf::Color(gray, gray, gray, originalColor.a);
322 }
323
324 default:
325 return originalColor;
326 }
