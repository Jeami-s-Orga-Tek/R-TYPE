47template <typename T>
48void Engine::ComponentManager::registerComponent()
49{
50 const std::string type_name = typeid(T).name();
51
52 if (component_types.find(type_name) != component_types.end())
53 throw Engine::ComponentManagerError("Trying to register component already in component manager !!!!");
54
55 component_types.insert({type_name, next_component_type});
56 component_arrays.insert({type_name, std::make_shared<Engine::ComponentArray<T>>()});
57
58 next_component_type++;
59}
