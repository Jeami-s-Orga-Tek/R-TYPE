111void RTypeServer::Server::gameLoop()
112{
113 if (!networkManager)
114 throw std::runtime_error("Network manager not initialized in game loop :(");
115 if (!networkManager->mediator)
116 throw std::runtime_error("Mediator not initialized in game loop :(");
117
118 std::cout << "NetworkManager server started. Press Ctrl+C to exit." << std::endl;
119
120 const float FIXED_DT = 1.0f / 60.0f;
121 const auto FRAME_DURATION = std::chrono::duration_cast<std::chrono::microseconds>(
122 std::chrono::duration<float>(FIXED_DT));
123
124 float accumulator = 0.0f;
125 auto previous_time = std::chrono::high_resolution_clock::now();
126
127 bool have_players_spawned = false;
128
129 std::shared_ptr<Engine::Mediator> mediator = networkManager->mediator;
130
131 while (true) {
132 auto frame_start = std::chrono::high_resolution_clock::now();
133 auto current_time = frame_start;
134 float frame_time = std::chrono::duration<float>(current_time - previous_time).count();
135 previous_time = current_time;
136 accumulator += frame_time;
137
138 while (accumulator >= FIXED_DT) {
139 if (!have_players_spawned && networkManager->getConnectedPlayers() >= player_nb) {
140 for (int i = 0; i < networkManager->getConnectedPlayers(); i++)
141 createPlayer();
142 createEnemy(1000, rand() % 400, ENEMY_TYPES::SIMPLE);
143 createBackground();
144 have_players_spawned = true;
145 }
146
147 if (have_players_spawned && rand() % 100 == 0) {
148 createEnemy(1000, rand() % 400, static_cast<ENEMY_TYPES>(rand() % 2));
149 }
150
151 player_control_system->update(networkManager, FIXED_DT);
152 physics_system->update(mediator, FIXED_DT);
153 enemy_system->update(networkManager, FIXED_DT);
154 collision_system->update(networkManager);
155
156 networkManager->handleTimeouts();
157
158 accumulator -= FIXED_DT;
159 }
160
161 auto frame_end = std::chrono::high_resolution_clock::now();
162 auto frame_elapsed = frame_end - frame_start;
163 if (frame_elapsed < FRAME_DURATION) {
164 std::this_thread::sleep_for(FRAME_DURATION - frame_elapsed);
165 }
166 }
167}
