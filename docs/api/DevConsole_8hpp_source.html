58 void treatNewTextInput(Event &event) {
59 uint32_t keycode = event.getParam<uint32_t>(0);
60 uint32_t unicode = event.getParam<uint32_t>(1);
61
62 // std::cout << keycode << " | " << unicode << std::endl;
63
64 if (unicode == '\b') {
65 if (!current_command.empty() && cursor_pos > 0) {
66 current_command.erase(cursor_pos - 1, 1);
67 cursor_pos--;
68 }
69 }
70 else if (unicode == 127) {
71 if (!current_command.empty() && cursor_pos < current_command.size()) {
72 current_command.erase(cursor_pos, 1);
73 }
74 }
75 else if (unicode == '\r' || unicode == '\n') {
76 old_std_out = "";
77 command_to_parse = current_command;
78 old_std_out += current_command;
79 old_std_out += "\n";
80 current_command = "";
81 cursor_pos = 0;
82 }
83 else if (unicode < 128 && std::isprint(unicode)) {
84 current_command.insert(cursor_pos, 1, static_cast<char>(unicode));
85 cursor_pos++;
86 }
87
88 if (keycode == static_cast<uint32_t>(KeyCodes::LEFT) && cursor_pos > 0) {
89 cursor_pos--;
90 }
91 if (keycode == static_cast<uint32_t>(KeyCodes::RIGHT) && cursor_pos < current_command.size()) {
92 cursor_pos++;
93 }

96 void parseCommand(std::shared_ptr<NetworkManager> networkManager, std::shared_ptr<Renderer> renderer) {
97 if (command_to_parse.rfind("set ", 0) == 0) {
98 size_t first_space = command_to_parse.find(' ');

100 if (first_space != std::string::npos && second_space != std::string::npos) {
101 std::string var = command_to_parse.substr(first_space + 1, second_space - first_space - 1);
102 std::string value = command_to_parse.substr(second_space + 1);
103 console_vars[var] = value;
104 old_std_out += "\tSet var '" + var + "' to '" + value + "'\n";
105 } else {
106 old_std_out += "\tUsage: set <var> <value>\n";

108 } else if (command_to_parse.rfind("get ", 0) == 0) {
109 size_t first_space = command_to_parse.find(' ');

112 if (it != console_vars.end()) {
113 old_std_out += "\t" + var + " = '" + it->second + "'\n";
114 } else {
115 old_std_out += "\tVar '" + var + "' not found\n";
116 }
117 } else if (command_to_parse.rfind("get_entity ", 0) == 0) {
118 size_t first_space = command_to_parse.find(' ');
119 Entity entity = std::stoi(command_to_parse.substr(first_space + 1));

131 } else if (command_to_parse.rfind("set_entity ", 0) == 0) {
132 try {
133 size_t first_space = command_to_parse.find(' ');
134 Entity entity = std::stoi(command_to_parse.substr(first_space + 1));
135 std::string component_name = "";

139 component_name = command_to_parse.substr(second_space + 1, third_space - second_space - 1);

141 std::string value = command_to_parse.substr(command_to_parse.find(' ', third_space + 1) + 1);
142
143 //TEMP

154 } else {
155 old_std_out += "\tUsage: set_entity <entity> <component> <field> <value>\n";

157 } catch (...) {}
158 } else if (command_to_parse == "list_vars") {
159 old_std_out += "\tConsole Vars:\n";
160 for (const auto &pair : console_vars) {
161 old_std_out += "\t " + pair.first + " = '" + pair.second + "'\n";

183 template <typename T>
184 void reflectComponent(std::shared_ptr<NetworkManager> networkManager, Entity entity) {
185 try {
186 const auto &comp = networkManager->mediator->getComponent<T>(entity);
187 const auto &field_names = boost::pfr::names_as_array<typeof(comp)>();
188 old_std_out += "\t" + boost::typeindex::type_id<T>().pretty_name() + " :\n";
189 uint8_t index = 0;
190 boost::pfr::for_each_field(comp, [this, field_names, &index](const auto &field) {
191 std::ostringstream oss;
192 oss << field_names[index] << " : " << field;
193 old_std_out += "\t\t" + oss.str() + "\n";
194 index++;
195 });
196 } catch (...) {}
197 }

200 void setComponentValue(std::shared_ptr<NetworkManager> networkManager, Entity entity, const std::string &component_name, const std::string &field_name, std::string value) {
201 try {
202 if (component_name != boost::typeindex::type_id<T>().pretty_name())

206 const auto &field_names = boost::pfr::names_as_array<typeof(comp)>();
207 uint8_t index = 0;
208 boost::pfr::for_each_field(comp, [this, &field_names, &index, &field_name, &value](auto &field) {
209 if (field_names[index] == field_name) {
210 // std::stringstream(value) >> value;
211 // field = std::any_cast<decltype(field)>(value);
212
213 //UGLY
214 if constexpr (std::is_arithmetic_v<std::remove_reference_t<decltype(field)>>) {
215 field = boost::lexical_cast<std::remove_reference_t<decltype(field)>>(value);
216 } else {
217 std::cout << "Custom type assignment not implemented for this field." << std::endl;
218 }
219 return;
220 }
221 index++;
222 });
223 } catch (const std::exception &e) {
224 std::cerr << e.what() << std::endl;
225 }
