55void ParticleSystem::update(float deltaTime)
56{
57 spawnTimer += deltaTime;
58 if (spawnTimer >= 1.0f / spawnRate && particles.size() < static_cast<size_t>(maxParticles)) {
59 spawnTimer = 0.0f;
60
61 float rand = distribution(generator);
62 if (rand < 0.6f) {
63 spawnParticle(ParticleType::STAR);
64 } else if (rand < 0.85f) {
65 spawnParticle(ParticleType::DUST);
66 } else if (rand < 0.98f) {
67 spawnParticle(ParticleType::NEBULA);
68 } else {
69 spawnParticle(ParticleType::SHOOTING_STAR);
70 }
71 }
72
73 for (auto it = particles.begin(); it != particles.end();) {
74 updateParticle(*it, deltaTime);
75 if (it->life <= 0.0f ||
76 it->position.x < -50 || it->position.x > windowSize.x + 50 ||
77 it->position.y < -50 || it->position.y > windowSize.y + 50) {
78 it = particles.erase(it);
79 } else {
80 ++it;
81 }
82 }
83}

93{
94 Engine::Utils::Vec2UInt oldSize = windowSize;

96
97 if (oldSize.x > 0 && oldSize.y > 0) {

99 float scaleY = static_cast<float>(newSize.y) / static_cast<float>(oldSize.y);
100
101 for (auto& particle : particles) {

104 particle.position.x = std::max(0.0f, std::min(static_cast<float>(newSize.x), particle.position.x));
105 particle.position.y = std::max(0.0f, std::min(static_cast<float>(newSize.y), particle.position.y));

110void ParticleSystem::spawnParticle(ParticleType type)
111{
112 sf::Vector2f position = getRandomPosition();
113 sf::Vector2f velocity = getRandomVelocity(type);
114 sf::Color color = getRandomColor(type);
115 float size = getRandomSize(type);
116 float life = getRandomLife(type);
117
118 Particle particle(position, velocity, color, size, life, type);
119
120 switch (type) {
121 case ParticleType::STAR:
122 particle.scintillation = distribution(generator) * 6.28f; // 2Ï€
123 break;
124 case ParticleType::SHOOTING_STAR:
125 particle.velocity.x = -200.0f - distribution(generator) * 300.0f;
126 particle.velocity.y = 50.0f + distribution(generator) * 100.0f;
127 particle.position.x = windowSize.x + 50;
128 particle.position.y = distribution(generator) * windowSize.y * 0.3f;
129 break;
130 case ParticleType::NEBULA:
131 particle.rotationSpeed = (distribution(generator) - 0.5f) * 2.0f;
132 break;
133 case ParticleType::DUST:
134 particle.alpha = 100.0f + distribution(generator) * 50.0f;
135 break;
136 }
137
138 particles.push_back(particle);

141void ParticleSystem::updateParticle(Particle& particle, float deltaTime)
142{
143 particle.position += particle.velocity * deltaTime;
144 particle.life -= deltaTime;
145 particle.rotation += particle.rotationSpeed * deltaTime;
146
147 switch (particle.type) {
148 case ParticleType::STAR:
149 particle.scintillation += deltaTime * 3.0f;
150 particle.alpha = 200.0f + 55.0f * std::sin(particle.scintillation);
151 break;
152
153 case ParticleType::DUST:
154 particle.velocity.y += std::sin(particle.position.x * 0.01f) * 10.0f * deltaTime;
155 particle.alpha = (particle.life / particle.maxLife) * 120.0f;
156 break;
157
158 case ParticleType::NEBULA:
159 particle.size += deltaTime * 5.0f;
160 particle.alpha = (particle.life / particle.maxLife) * 60.0f;
161 break;
162 case ParticleType::SHOOTING_STAR:
163 particle.alpha = (particle.life / particle.maxLife) * 255.0f;
164 break;
165 }
166 particle.color.a = static_cast<sf::Uint8>(std::max(0.0f, std::min(255.0f, particle.alpha)));
167}

169void ParticleSystem::renderParticle(sf::RenderWindow& window, const Particle& particle)
170{
171 sf::CircleShape shape;
172
173 switch (particle.type) {
174 case ParticleType::STAR:
175 shape.setRadius(particle.size);
176 shape.setPointCount(5);
177 break;
178
179 case ParticleType::SHOOTING_STAR:
180 shape.setRadius(particle.size);
181 shape.setScale(3.0f, 0.5f);
182 break;
183
184 default:
185 shape.setRadius(particle.size);
186 shape.setPointCount(30);
187 break;
188 }
189 sf::Color finalColor = particle.color;
190 if (parameters) {
191 finalColor = parameters->applyColorBlindFilter(particle.color);
192 }
193
194 shape.setFillColor(finalColor);
195 shape.setPosition(particle.position.x - particle.size, particle.position.y - particle.size);
196 shape.setRotation(particle.rotation);
197
198 window.draw(shape);
199
200 if (particle.type == ParticleType::STAR && particle.size > 2.0f) {
201 sf::CircleShape halo;
202 halo.setRadius(particle.size * 2.0f);
203 halo.setPointCount(30);
204 sf::Color haloColor = particle.color;
205 haloColor.a = static_cast<sf::Uint8>(haloColor.a * 0.2f);
206
207 if (parameters) {
208 haloColor = parameters->applyColorBlindFilter(haloColor);
209 }
210 halo.setFillColor(haloColor);
211 halo.setPosition(particle.position.x - particle.size * 2.0f,
212 particle.position.y - particle.size * 2.0f);
213 window.draw(halo);
214 }
