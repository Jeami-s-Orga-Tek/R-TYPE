30bool NetworkClient::connect(const std::string& serverIP, uint16_t port, const std::string& playerName)
31{
32 try {
33 m_io = std::make_unique<boost::asio::io_context>();
34 m_socket = std::make_unique<udp::socket>(*m_io);
35 m_serverEndpoint = udp::endpoint(boost::asio::ip::address::from_string(serverIP), port);
36 m_socket->open(udp::v4());
37 m_playerName = playerName;
38 m_running = true;
39
40 startReceive();
41 m_networkThread = std::make_unique<std::thread>([this]() { networkThread(); });
42 std::vector<uint8_t> out;
43 NetHeader h{};
44 h.magic = kMagic;
45 h.version = kProtoVersion;
46 h.type = static_cast<uint8_t>(MsgType::HELLO);
47 m_protocol.writeHeader(out, h);
48 HelloBody hb{};
49 hb.client_nonce = 0;
50 hb.proto = kProtoVersion;
51 hb.name = m_playerName;
52 m_protocol.writeHelloBody(out, hb);
53 sendPacket(out);
54
55 m_connected = true;
56 return true;
57 } catch (const std::exception& e) {
58 std::cerr << "NetworkClient connect exception: " << e.what() << std::endl;
59 return false;
60 }

166void NetworkClient::processMessage(const uint8_t* data, std::size_t size)
167{
168 try {
169 NetHeader h = m_protocol.readHeader(data, size);
170 if (h.magic != kMagic || h.version != kProtoVersion) {
171 std::cerr << "Bad packet received (magic/version mismatch)" << std::endl;
172 return;
173 }
174 std::size_t off = sizeof(NetHeader);
175 MsgType t = static_cast<MsgType>(h.type);
176 switch (t) {
177 case MsgType::WELCOME: {
178 WelcomeBody wb = m_protocol.readWelcomeBody(data, size, off);
179 m_playerId = wb.playerId;
180 m_connected = true;
181 if (m_onConnected) m_onConnected(m_playerId);
182 listRooms();
183 break;
184 }
185 case MsgType::ROOM_CREATED: {
186 RoomCreatedBody rb = m_protocol.readRoomCreatedBody(data, size, off);
187 if (m_onRoomCreated) m_onRoomCreated(rb);
188 break;
189 }
190 case MsgType::ROOM_JOINED: {
191 RoomJoinedBody rb = m_protocol.readRoomJoinedBody(data, size, off);
192 if (m_onRoomJoined) m_onRoomJoined(rb);
193 listRooms();
194 break;
195 }
196 case MsgType::ROOM_LEFT: {
197 RoomLeftBody rb = m_protocol.readRoomLeftBody(data, size, off);
198 if (m_onRoomLeft) m_onRoomLeft(rb);
199 listRooms();
200 break;
201 }
202 case MsgType::ROOM_LIST: {
203 RoomListBody rb = m_protocol.readRoomListBody(data, size, off);
204 if (m_onRoomList) m_onRoomList(rb);
205 break;
206 }
207 case MsgType::ROOM_MESSAGE: {
208 RoomMessageBody rb = m_protocol.readRoomMessageBody(data, size, off);
209 if (m_onRoomMessage) m_onRoomMessage(rb);
210 break;
211 }
212 case MsgType::ROOM_ERROR: {
213 RoomErrorBody rb = m_protocol.readRoomErrorBody(data, size, off);
214 if (m_onRoomError) m_onRoomError(rb);
215 break;
216 }
217 default:
218 break;
219 }
220 } catch (const std::exception& e) {
221 std::cerr << "processMessage exception: " << e.what() << std::endl;
222 }
223}
