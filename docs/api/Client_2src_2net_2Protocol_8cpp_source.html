30NetHeader Protocol::readHeader(const uint8_t* data, std::size_t size)
31{
32 NetHeader h{};
33
34 if (size < sizeof(NetHeader))
35 throw std::runtime_error("packet too small for header");
36 std::size_t off = 0;
37 h.magic = read_u16be(data, off, size);
38 h.version = read_u8(data, off, size);
39 h.type = read_u8(data, off, size);
40 h.seq = read_u32be(data, off, size);
41 h.ack = read_u32be(data, off, size);
42 h.ackBits = read_u32be(data, off, size);
43 return h;
44}

96void Protocol::writeHelloBody (std::vector<uint8_t>& out, const HelloBody& b)
97{
98 append_u32be(out, b.client_nonce);
99 append_u8(out, b.proto);
100 uint8_t len = static_cast<uint8_t>(std::min<std::size_t>(255, b.name.size()));
101 append_u8(out, len);
102 out.insert(out.end(), b.name.data(), b.name.data()+len);
103}

105HelloBody Protocol::readHelloBody(const uint8_t* data, std::size_t size, std::size_t& off)
106{
107 HelloBody b{};

109 b.proto = read_u8(data, off, size);

235void Protocol::writeRoomListBody(std::vector<uint8_t>& out, const RoomListBody& b)
236{
237 append_u16be(out, static_cast<uint16_t>(b.rooms.size()));
238 for (const auto& room : b.rooms) {
239 append_u32be(out, room.roomId);
240 writeString(out, room.roomName);
241 append_u8(out, room.currentPlayers);
242 append_u8(out, room.maxPlayers);
243 }
244}

246RoomListBody Protocol::readRoomListBody(const uint8_t* data, std::size_t size, std::size_t& off)
247{
248 RoomListBody b{};
249 uint16_t roomCount = read_u16be(data, off, size);
250 b.rooms.reserve(roomCount);
251 for (uint16_t i = 0; i < roomCount; ++i) {
252 RoomInfoEntry entry{};
253 entry.roomId = read_u32be(data, off, size);
254 entry.roomName = readString(data, size, off);
255 entry.currentPlayers = read_u8(data, off, size);
256 entry.maxPlayers = read_u8(data, off, size);
257 b.rooms.push_back(entry);
258 }
259 return b;
260}
