75void Engine::Renderers::SFML::handleEvents(std::shared_ptr<Engine::NetworkManager> networkManager)
76{
77 std::bitset<5> buttons {};
78 sf::Event event;
79
80 while (window->pollEvent(event)) {
81 if (event.type == sf::Event::Closed)
82 window->close();
83 if (event.type == sf::Event::KeyPressed) {
84 if (event.key.code == sf::Keyboard::Space && window->hasFocus()) {
85 buttons.set(static_cast<std::size_t>(Engine::InputButtons::SHOOT));
86 }
87 if (event.key.code == sf::Keyboard::Equal && window->hasFocus()) {
88 is_console_open = !is_console_open;
89 Engine::Event dev_console_input_event(static_cast<Engine::EventId>(Engine::EventsIds::DEVCONSOLE_KEY_PRESSED));
90 networkManager->mediator->sendEvent(dev_console_input_event);
91 continue;
92 }
93
94 // std::cout << event.key.code << " " << event.key.scancode << " " << event.text.unicode << " " << sf::Keyboard::getDescription(event.key.scancode).toAnsiString() << std::endl;
95
96 uint32_t keycode = static_cast<uint32_t>(event.key.code);
97
98 switch (event.key.code) {
99 case sf::Keyboard::Left:
100 keycode = static_cast<uint32_t>(KeyCodes::LEFT);
101 break;
102 case sf::Keyboard::Right:
103 keycode = static_cast<uint32_t>(KeyCodes::RIGHT);
104 break;
105 case sf::Keyboard::Up:
106 keycode = static_cast<uint32_t>(KeyCodes::UP);
107 break;
108 case sf::Keyboard::Down:
109 keycode = static_cast<uint32_t>(KeyCodes::DOWN);
110 break;
111 // case sf::Keyboard::Delete:
112 // keycode = static_cast<uint32_t>(KeyCodes::DELETE_KEY);
113 // break;
114 default:
115 continue;
116 }
117
118 Engine::Event dev_console_text_entered_event(static_cast<Engine::EventId>(Engine::EventsIds::DEVCONSOLE_TEXT_ENTERED));
119
120 dev_console_text_entered_event.setParam(0, keycode);
121 dev_console_text_entered_event.setParam(1, static_cast<uint32_t>(0));
122 networkManager->mediator->sendEvent(dev_console_text_entered_event);
123 }
124 if (event.type == sf::Event::TextEntered) {
125 if (!is_console_open || event.text.unicode == '=')
126 continue;
127
128 Engine::Event dev_console_text_entered_event(static_cast<Engine::EventId>(Engine::EventsIds::DEVCONSOLE_TEXT_ENTERED));
129
130 uint32_t keycode = static_cast<uint32_t>(event.key.code);
131 uint32_t unicode = static_cast<uint32_t>(event.text.unicode);
132
133 dev_console_text_entered_event.setParam(0, keycode);
134 dev_console_text_entered_event.setParam(1, unicode);
135 networkManager->mediator->sendEvent(dev_console_text_entered_event);
136 }
137 }
138
139 //Don't take player input when console is open (also pause and menus)
140 if (is_console_open)
141 return;
142
143 if (window->hasFocus() && sf::Joystick::isConnected(0)) {
144 float x = sf::Joystick::getAxisPosition(0, sf::Joystick::X);
145 float y = sf::Joystick::getAxisPosition(0, sf::Joystick::Y);
146
147 const float threshold = 30.f;
148
149 if (x < -threshold)
150 buttons.set(static_cast<std::size_t>(Engine::InputButtons::LEFT));
151 if (x > threshold)
152 buttons.set(static_cast<std::size_t>(Engine::InputButtons::RIGHT));
153 if (y < -threshold)
154 buttons.set(static_cast<std::size_t>(Engine::InputButtons::UP));
155 if (y > threshold)
156 buttons.set(static_cast<std::size_t>(Engine::InputButtons::DOWN));
157
158 if (sf::Joystick::isButtonPressed(0, 0)) {
159 buttons.set(static_cast<std::size_t>(Engine::InputButtons::SHOOT));
160 }
161 }
162
163 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Left) || sf::Keyboard::isKeyPressed(sf::Keyboard::Q))) {
164 buttons.set(static_cast<std::size_t>(Engine::InputButtons::LEFT));
165 }
166 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Right) || sf::Keyboard::isKeyPressed(sf::Keyboard::D))) {
167 buttons.set(static_cast<std::size_t>(Engine::InputButtons::RIGHT));
168 }
169 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Up) || sf::Keyboard::isKeyPressed(sf::Keyboard::Z))) {
170 buttons.set(static_cast<std::size_t>(Engine::InputButtons::UP));
171 }
172 if (window->hasFocus() && (sf::Keyboard::isKeyPressed(sf::Keyboard::Down) || sf::Keyboard::isKeyPressed(sf::Keyboard::S))) {
173 buttons.set(static_cast<std::size_t>(Engine::InputButtons::DOWN));
174 }
175
176 Engine::Event player_input_event(static_cast<Engine::EventId>(Engine::EventsIds::PLAYER_INPUT));
177 player_input_event.setParam(0, networkManager->player_id);
178 player_input_event.setParam(1, buttons);
179 networkManager->mediator->sendEvent(player_input_event);
180 networkManager->sendInput(networkManager->player_id, networkManager->room_id, buttons);
181}

183bool Engine::Renderers::SFML::loadTexture(const std::string& id, const std::string& filepath) {
184 sf::Texture texture;
185
186 if (filepath.empty()) {
187 sf::Image whiteImage;
188 whiteImage.create(1, 1, sf::Color::White);
189 if (!texture.loadFromImage(whiteImage))
190 return (false);
191 } else {
192 if (!texture.loadFromFile(filepath))
193 return (false);
194 }
195
196 texture.setRepeated(true);
197 textures[id] = std::move(texture);
198 return (true);
199}

302void Engine::Renderers::SFML::drawText(const std::string& fontId, const std::string& text, float x, float y, unsigned int size, unsigned int color)
303{
304 if (!window)
305 return;
306 auto it = fonts.find(fontId);
307 if (it == fonts.end())
308 return;
309 sf::Text sfText;
310 sfText.setFont(it->second);
311 sfText.setString(text);
312 sfText.setCharacterSize(size);
313 sfText.setPosition(x, y);
314 sfText.setFillColor(sf::Color(
315 (color >> 24) & 0xFF,
316 (color >> 16) & 0xFF,
317 (color >> 8) & 0xFF,
318 color & 0xFF
319 ));
320 window->draw(sfText);
321}
