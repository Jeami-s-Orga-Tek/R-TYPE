51template <typename T>
52void Engine::ComponentArray<T>::insertEntity(Engine::Entity entity, T component)
53{
54 if (entity_to_index_map.find(entity) != entity_to_index_map.end())
55 throw Engine::ComponentArrayError("Entity already in component array !!!!!!");
56
57 entity_to_index_map[entity] = size;
58 index_to_entity_map[size] = entity;
59 components[size] = component;
60 size++;
61}

63template <typename T>
64void Engine::ComponentArray<T>::removeEntity(Engine::Entity entity)
65{
66 if (entity_to_index_map.find(entity) == entity_to_index_map.end())
67 throw Engine::ComponentArrayError("Trying to delete entity not in component array !!!!!!");
68
69 std::size_t removed_entity_index = entity_to_index_map[entity];
70 std::size_t index_of_last_elem = size - 1;
71
72 components[removed_entity_index] = components[index_of_last_elem];
73
74 Engine::Entity entity_of_last_elem = index_to_entity_map[index_of_last_elem];
75 entity_to_index_map[entity_of_last_elem] = removed_entity_index;
76 index_to_entity_map[removed_entity_index] = entity_of_last_elem;
77
78 entity_to_index_map.erase(entity);
79 index_to_entity_map.erase(index_of_last_elem);
80 size--;
81}
