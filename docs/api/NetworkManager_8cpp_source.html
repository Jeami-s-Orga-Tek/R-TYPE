98void Engine::NetworkManager::handle_receive(std::size_t bytes_recvd)
99{
100 if (bytes_recvd < 16)
101 return;
102 PacketHeader header;
103 std::memcpy(&header, recv_buffer.data(), sizeof(header));
104 header.magic = ntohs(header.magic);
105 header.seq = ntohl(header.seq);
106 header.ack = ntohl(header.ack);
107 header.ack_bits = ntohl(header.ack_bits);
108
109 // std::cout << "RECEIVE " << (int)header.type << std::endl;
110 if (header.magic != MAGIC || header.version != PROTO_VERSION)
111 return;
112 if (role == Role::SERVER) {
113 switch (header.type) {
114 case MSG_HELLO: {
115 auto it = std::find_if(client_endpoints.begin(), client_endpoints.end(),
116 [this](const boost::asio::ip::udp::endpoint &ep) {
117 return ep == remote_endpoint;
118 });
119 if (it == client_endpoints.end()) {
120 client_endpoints.push_back(remote_endpoint);
121 }
122 send_welcome();
123 break;
124 }
125 case MSG_PING:
126 send_pong(header.seq);
127 break;
128 case MSG_PONG:
129 send_ping(header.seq);
130 break;
131 case MSG_INPUT:
132 receiveInputs();
133 break;
134 default:
135 break;
136 }
137 } else {
138 switch (header.type) {
139 case MSG_WELCOME:
140 receiveWelcome();
141 break;
142 case MSG_ENTITY:
143 receiveEntity();
144 break;
145 case MSG_ENTITY_DESTROYED:
146
147 break;
148 case MSG_COMPONENT:
149 receiveComponent();
150 break;
151 default:
152 break;
153 }
154 }
155}

175void Engine::NetworkManager::send_hello(const std::string &client_name, uint32_t nonce)
176{
177 std::string name = client_name.substr(0, 255);
178 size_t name_len = name.size();
179 size_t total_size = sizeof(PacketHeader) + 6 + name_len;
180 std::vector<uint8_t> buf(total_size);
181
182 const PacketHeader &ph = createPacketHeader(MSG_HELLO);
183 std::memcpy(buf.data(), &ph, sizeof(ph));
184
185 size_t offset = sizeof(ph);
186 uint32_t nonce_n = htonl(nonce);
187 buf[offset + 0] = (nonce_n >> 24) & 0xFF;
188 buf[offset + 1] = (nonce_n >> 16) & 0xFF;
189 buf[offset + 2] = (nonce_n >> 8) & 0xFF;
190 buf[offset + 3] = (nonce_n) & 0xFF;
191 buf[offset + 4] = PROTO_VERSION;
192 buf[offset + 5] = static_cast<uint8_t>(name_len);
193 std::memcpy(buf.data() + offset + 6, name.data(), name_len);
194
195 socket.send_to(boost::asio::buffer(buf.data(), total_size), remote_endpoint);
196}
