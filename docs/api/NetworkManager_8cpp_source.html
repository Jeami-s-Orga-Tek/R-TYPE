97void Engine::NetworkManager::handle_receive(std::size_t bytes_recvd)
98{
99 if (bytes_recvd < 16)
100 return;
101 PacketHeader header;
102 std::memcpy(&header, recv_buffer.data(), sizeof(header));
103 header.magic = ntohs(header.magic);
104 header.seq = ntohl(header.seq);
105 header.ack = ntohl(header.ack);
106 header.ack_bits = ntohl(header.ack_bits);
107
108 // std::cout << "RECEIVE " << (int)header.type << std::endl;
109 if (header.magic != MAGIC || header.version != PROTO_VERSION)
110 return;
111 if (role == Role::SERVER) {
112 switch (header.type) {
113 case MSG_HELLO: {
114 auto it = std::find_if(client_endpoints.begin(), client_endpoints.end(),
115 [this](const std::pair<const long unsigned int, boost::asio::ip::basic_endpoint<boost::asio::ip::udp>> &ep) {
116 return ep.second == remote_endpoint;
117 });
118 if (it == client_endpoints.end()) {
119 // client_endpoints.push_back(remote_endpoint);
120 client_endpoints[client_endpoints.size()] = remote_endpoint;
121 // client_timeout_clocks.push_back(std::chrono::steady_clock::now());
122 client_timeout_clocks[remote_endpoint] = std::chrono::steady_clock::now();
123 }
124 sendWelcome();
125 break;
126 }
127 case MSG_PING:
128 sendPong(header.seq);
129 break;
130 case MSG_PONG:
131 sendPing(header.seq);
132 break;
133 case MSG_INPUT:
134 receiveInputs();
135 break;
136 default:
137 break;
138 }
139 } else {
140 switch (header.type) {
141 case MSG_WELCOME:
142 receiveWelcome();
143 break;
144 case MSG_ENTITY:
145 receiveEntity();
146 break;
147 case MSG_ENTITY_DESTROYED:
148 receiveDestroyEntity();
149 break;
150 case MSG_COMPONENT:
151 receiveComponent();
152 break;
153 default:
154 break;
155 }
156 }
157}

207void Engine::NetworkManager::sendHello(const std::string &client_name, uint32_t nonce)
208{
209 std::string name = client_name.substr(0, 255);
210 name.resize(255, '\0');
211 size_t name_len = name.size();
212 size_t total_size = sizeof(PacketHeader) + 6 + name_len;
213 std::vector<uint8_t> buf(total_size);
214
215 const PacketHeader &ph = createPacketHeader(MSG_HELLO);
216 std::memcpy(buf.data(), &ph, sizeof(ph));
217
218 size_t offset = sizeof(ph);
219 uint32_t nonce_n = htonl(nonce);
220 buf[offset + 0] = (nonce_n >> 24) & 0xFF;
221 buf[offset + 1] = (nonce_n >> 16) & 0xFF;
222 buf[offset + 2] = (nonce_n >> 8) & 0xFF;
223 buf[offset + 3] = (nonce_n) & 0xFF;
224 buf[offset + 4] = PROTO_VERSION;
225 buf[offset + 5] = static_cast<uint8_t>(name_len);
226 std::memcpy(buf.data() + offset + 6, name.data(), name_len);
227
228 socket.send_to(boost::asio::buffer(buf.data(), total_size), remote_endpoint);
229}
