145template <std::size_t S>
146void Engine::NetworkManager::sendInput(const uint32_t player_id, const uint16_t room_id, const std::bitset<S> inputs)
147{
148 std::array<uint8_t, sizeof(Engine::NetworkManager::PacketHeader) + sizeof(Engine::NetworkManager::InputBody)> buf {};
149 const Engine::NetworkManager::PacketHeader &ph = createPacketHeader(MSG_INPUT);
150 std::memcpy(buf.data(), &ph, sizeof(ph));
151
152 InputBody ib = {
153 .player_id = htonl(player_id),
154 .room_id = room_id,
155 .input_data = inputs.to_ullong(),
156 };
157 std::memcpy(buf.data() + sizeof(ph), &ib, sizeof(ib));
158 socket.send_to(boost::asio::buffer(buf), remote_endpoint);
159}

173template <typename T>
174void Engine::NetworkManager::sendComponent(const Entity &entity, const T &component)
175{
176 const std::string type_name = typeid(T).name();
177 size_t total_size = sizeof(PacketHeader) + sizeof(ComponentBody) + type_name.size() + sizeof(component);
178 std::vector<uint8_t> buf(total_size);
179
180 const PacketHeader &ph = createPacketHeader(MSG_COMPONENT);
181 std::memcpy(buf.data(), &ph, sizeof(ph));
182
183 ComponentBody cb = {
184 .entity_id = entity,
185 .name_len = static_cast<uint8_t>(type_name.length()),
186 .component_len = static_cast<uint32_t>(sizeof(component)),
187 };
188 std::memcpy(buf.data() + sizeof(ph), &cb, sizeof(cb));
189 std::memcpy(buf.data() + sizeof(ph) + sizeof(cb), type_name.data(), cb.name_len);
190 std::memcpy(buf.data() + sizeof(ph) + sizeof(cb) + cb.name_len, &component, sizeof(component));
191
192 if (role == Role::SERVER) {
193 for (const auto &endpoint : client_endpoints) {
194 socket.send_to(boost::asio::buffer(buf.data(), buf.size()), endpoint.second);
195 }
196 } else {
197 socket.send_to(boost::asio::buffer(buf.data(), buf.size()), remote_endpoint);
198 }
199}
