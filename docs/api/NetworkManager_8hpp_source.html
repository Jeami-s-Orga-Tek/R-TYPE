158template <std::size_t S>
159void Engine::NetworkManager::sendInput(const uint32_t player_id, const uint16_t room_id, const std::bitset<S> inputs)
160{
161 std::array<uint8_t, sizeof(Engine::NetworkManager::PacketHeader) + sizeof(Engine::NetworkManager::InputBody)> buf {};
162 const Engine::NetworkManager::PacketHeader &ph = createPacketHeader(MSG_INPUT);
163 std::memcpy(buf.data(), &ph, sizeof(ph));
164
165 InputBody ib = {
166 .player_id = htonl(player_id),
167 .room_id = room_id,
168 .input_data = inputs.to_ullong(),
169 };
170 std::memcpy(buf.data() + sizeof(ph), &ib, sizeof(ib));
171 socket.send_to(boost::asio::buffer(buf), remote_endpoint);
172}
